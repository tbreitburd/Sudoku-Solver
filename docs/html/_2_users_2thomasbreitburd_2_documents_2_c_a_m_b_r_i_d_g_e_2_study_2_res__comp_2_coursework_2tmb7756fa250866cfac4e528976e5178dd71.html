145def box(sudoku, row, col):
146 """!@brief This function takes in a sudoku and a cell's row and column,
147 and returns a 3x3 array of the sudoku box that cell is in.
148
149 @details The function takes in a sudoku and a cell's row and column,
150 and returns a 3x3 array of the sudoku box that cell is in.
151
152 @param sudoku The sudoku to extract the box from
153 @param row The row of the cell, 0 based index
154 @param col The column of the cell, 0 based index
155
156 @return A 3x3 array of the corresponding box.
157
158 @exception ValueError Raised if the cell coordinates are not valid,
159 as they must be between 0 and 8 inclusive.
160
161 @example This example describes how to use the box function.
162
163 @code
164 sudoku = np.array([[0,0,4,0,5,0,0,7,0],
165 [0,0,0,0,0,0,0,0,0],
166 [0,0,0,0,0,8,0,0,0],
167 [0,3,0,0,0,0,0,9,0],
168 [0,0,1,0,0,0,0,0,0],
169 [0,0,0,7,0,0,0,0,0],
170 [0,0,0,4,6,2,0,0,0],
171 [0,8,0,0,0,0,0,0,0],
172 [0,7,0,0,0,0,3,0,0]])
173 box = box(sudoku, 7, 5)
174 print(box)
175 @endcode
176 """
177 # Check that the cell coordinates are valid
178 try:
179 if row < 0 or row > 8 or col < 0 or col > 8:
180 raise ValueError(
181 "The cell coordinates are not valid, "
182 + "must be between 0 and 8 inclusive but:"
183 + "Row: "
184 + str(row)
185 + ", Col: "
186 + str(col)
187 )
188 except ValueError as e:
189 print("Traceback: ")
190 traceback.print_stack()
191 print("Error: ")
192 print(e)
193 print("Exiting program")
194 sys.exit(1)
195
196 # Initialize box array
197 box = np.zeros((3, 3))
198 # Identify the box's box-coordinates (tuples of 1, 2 or 3)
199 box_row = int((row + 3) // 3)
200 box_col = int((col + 3) // 3)
201
202 # Get the box values
203 row_start = int(3 * (box_row - 1))
204 row_end = int(row_start + 3)
205
206 col_start = int(3 * (box_col - 1))
207 col_end = int(col_start + 3)
208
209 # fmt: off
210 # This line caused black/flake8 conflicts
211 box = sudoku[row_start:row_end, col_start:col_end]
212 # fmt: on

217def sudoku_to_output_format(sudoku):
218 """!@brief This function takes in a sudoku array,
219 and returns a string in the specific format of the output file.
220
221 @details This function takes in a sudoku numpy array, and adds each cell to
222 a string, adding in separators, vertical and horizontal
223 and newlines where necessary, returning a string which when printed will
224 be in the desired sudoku format for the output file.
225
226 @param sudoku The sudoku numpy array to convert to a string
227
228 @return A string in the format of the output file.
229
230 @example This example describes how to use the sudoku_to_output_format
231 function.
232
233 @code
234 sudoku = np.array([[0,0,4,0,5,0,0,7,0],
235 [0,0,0,0,0,0,0,0,0],
236 [0,0,0,0,0,8,0,0,0],
237 [0,3,0,0,0,0,0,9,0],
238 [0,0,1,0,0,0,0,0,0],
239 [0,0,0,7,0,0,0,0,0],
240 [0,0,0,4,6,2,0,0,0],
241 [0,8,0,0,0,0,0,0,0],
242 [0,7,0,0,0,0,3,0,0]])
243 sudoku_output = sudoku_to_output_format(sudoku)
244 print(sudoku_output)
245 @endcode
246 """
247 # Initialize the output string
248 sudoku_str = ""
249
250 for row in range(9):
251 for col in range(9):
252 # Add each cell to the output string
253 sudoku_str += str(sudoku[row][col])
254
255 # Add a vertical separator if at the end of a box
256 if col == 2 or col == 5:
257 sudoku_str += "|"
258
259 # Add a newline if at the end of a row,
260 # except at the end of the sudoku
261 elif col == 8 and row != 8:
262 sudoku_str += "\n"
263
264 # Add a horizontal separator if at the end of row 3 or 6
265 if row == 2 or row == 5:
266 sudoku_str += "---+---+---\n"
267
268 return sudoku_str
